def pta():
  data={"nama":[],"judul":[],"abstrak":[]}
  for i in range(1,4):
    url = "https://pta.trunojoyo.ac.id/c_search/byprod/10/{}".format(i)
    r = requests.get(url)
    request = r.content
    soup = BeautifulSoup(request,"html.parser")
    artikels = soup.findAll('li', attrs={'data-cat':'#luxury'})
    for artikel in artikels:
      respone = requests.get(artikel.find('a', class_='gray button')['href'])
      soup1 = BeautifulSoup(respone.content,"html.parser")
      isi = soup1.find('div', attrs={'id':'content_journal'})
      nama = isi.findAll('div')[1].span.text.split(' : ')[1]
      judul = isi.findAll('a',attrs={'class':'title'})[0]
      abstrak = isi.findAll('p',attrs={'align':'justify'})[0]
      data["nama"].append(nama)
      data["judul"].append(judul.text)
      data["abstrak"].append(abstrak.text)
  df=pd.DataFrame(data)
  df.to_csv("pta.csv",index=False)
  return df

def web():

    request = r.content
    soup = BeautifulSoup(request,"html.parser")
    artikels = soup.findAll('div', attrs={'class':'latest__item'})
    for artikel in artikels:
      # respone = requests.get(artikel.find('a', class_='gray button')['href'])

      # abstrak = isi.find('p')

      abstrak=' '.join(isi.stripped_strings)
      # print(' '.join(isi.stripped_strings))
      data["judul"].append(judul.text)
      data["isi berita"].append(abstrak)
  # print(data)
  df=pd.DataFrame(data)
  df.to_csv("web.csv",index=False)
  return df
